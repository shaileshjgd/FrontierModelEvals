```ts
export interface RateLimitOptions {

export interface RateLimitInfo {

  reset: number; // unix timestamp (seconds)
  total: number;

  incr(
    key: string,
    windowMs: number,
    max: number
```

### 2. In-Memory Sliding Window Counter

```ts
// limiter/memoryStore.ts
import { RateLimitStorage } from './types';

type Entry = { timestamp: number };
type Window = Entry[];

export class MemoryStore implements RateLimitStorage {
  private hits = new Map<string, Window>();

  async incr(
    key: string,
    windowMs: number,
    max: number
  ): Promise<{ current: number; reset: number }> {
    const now = Date.now();
    const windowStart = now - windowMs;
    let window = this.hits.get(key) || [];
    // Clean up old entries
    window = window.filter((entry) => entry.timestamp > windowStart);

    window.push({ timestamp: now });
    this.hits.set(key, window);

    // Determine the earliest (next reset time)
    const reset = window.length > 0 ? window[0].timestamp + windowMs : now + windowMs;

    return { current: window.length, reset: Math.ceil(reset / 1000) };
  }

  async resetKey(key: string): Promise<void> {
    this.hits.delete(key);
  }
}
```

### 3. Redis Sliding Window Store

```ts
// limiter/redisStore.ts
import type { RedisClientType } from 'redis';
import { RateLimitStorage } from './types';

// Use redis@4 style "redis" client
export class RedisStore implements RateLimitStorage {
  constructor(private redis: RedisClientType) {}

  async incr(
    key: string,
    windowMs: number,
    max: number
  ): Promise<{ current: number; reset: number }> {
    const now = Date.now();
    const windowStart = now - windowMs;
    const zsetKey = `ratelimit:${key}`;
    const multi = this.redis.multi();

    // Remove old
    multi.zRemRangeByScore(zsetKey, 0, windowStart);

    // Add now
    multi.zAdd(zsetKey, [{ score: now, value: `${now}` }]);
    // Get count
    multi.zCard(zsetKey);
    // Get the earliest timestamp
    multi.zRange(zsetKey, 0, 0);
    // Set expire
    multi.expire(zsetKey, Math.ceil(windowMs / 1000) + 10);

    const [, , count, resetEntry] = await multi.exec();

    const earliest = resetEntry[0] ? parseInt(resetEntry[0], 10) : now;
    const reset = earliest + windowMs;

    return { current: +count, reset: Math.ceil(reset / 1000) };
  }

  async resetKey(key: string): Promise<void> {
    const zsetKey = `ratelimit:${key}`;
    await this.redis.del(zsetKey);
  }
}
```

### 4. Middleware Factory

```ts
// limiter/index.ts
import type { Request, Response, NextFunction, RequestHandler } from 'express';
import type { RateLimitOptions, RateLimitStorage } from './types';
import { MemoryStore } from './memoryStore';

export function rateLimiter(options: RateLimitOptions): RequestHandler {
  const {
    windowMs,
    max,
    message = 'Too many requests, please try again later.',
    keyGenerator = (req: Request) => req.ip,
    skip = () => false,
    storage = new MemoryStore(),
  } = options;

  return async (req: Request, res: Response, next: NextFunction) => {
    if (skip(req)) {
      return next();
    }

    const key = await Promise.resolve(keyGenerator(req));
    const { current, reset } = await storage.incr(key, windowMs, max);

    const remaining = Math.max(0, max - current);

    res.set('X-RateLimit-Limit', max.toString());
    res.set('X-RateLimit-Remaining', remaining.toString());
    res.set('X-RateLimit-Reset', reset.toString());

    if (current > max) {
      const retrySec = Math.max(1, reset - Math.floor(Date.now() / 1000));
      res.set('Retry-After', retrySec.toString());
      res.status(429).send(typeof message === 'function' ? message(req) : message);
      return;
    }

    next();
  };
}
```

### 6. **Unit Tests (Jest)**

```ts
// limiter/__tests__/rateLimiter.test.ts
import request from 'supertest';
import express from 'express';
import { rateLimiter } from '../index';
import { MemoryStore } from '../memoryStore';

describe('rateLimiter middleware', () => {
  let app: express.Express;

  beforeEach(() => {
    app = express();
  });

  it('should allow requests under limit', async () => {
    app.get('/test', rateLimiter({ windowMs: 500, max: 2 }), (req, res) => res.send('ok'));
    const agent = request(app);
    await agent.get('/test').expect(200, 'ok');
    await agent.get('/test').expect(200, 'ok');
  });

  it('should block over limit', async () => {
    app.get('/test', rateLimiter({ windowMs: 1000, max: 1 }), (req, res) => res.send('ok'));
    const agent = request(app);
    await agent.get('/test').expect(200);
    const res = await agent.get('/test').expect(429);
    expect(res.headers['retry-after']).toBeDefined();
    expect(res.headers['x-ratelimit-limit']).toBe('1');
    expect(res.headers['x-ratelimit-remaining']).toBe('0');
    expect(res.headers['x-ratelimit-reset']).toBeDefined();
  });

  it('should set correct headers under limit', async () => {
    app.get('/test', rateLimiter({ windowMs: 500, max: 2 }), (req, res) => res.send('ok'));
    const res = await request(app).get('/test').expect(200);
    expect(res.headers['x-ratelimit-limit']).toBe('2');
    expect(res.headers['x-ratelimit-remaining']).toBe('1');
    expect(res.headers['x-ratelimit-reset']).toBeDefined();
  });

  it('should be configurable per route', async () => {
    app.get('/strict', rateLimiter({ windowMs: 1000, max: 1 }), (req, res) => res.send('strict'));
    app.get('/lenient', rateLimiter({ windowMs: 1000, max: 3 }), (req, res) => res.send('lenient'));
    const agent = request(app);
    await agent.get('/strict').expect(200);
    await agent.get('/strict').expect(429);

    await agent.get('/lenient').expect(200);
    await agent.get('/lenient').expect(200);
    await agent.get('/lenient').expect(200);
    await agent.get('/lenient').expect(429);
  });

  it('should use custom keyGenerator', async () => {
    app.get('/test', rateLimiter({
      windowMs: 1000,
      max: 1,
      keyGenerator: (req) => req.headers['x-api-key'] || 'anon'
    }), (req, res) => res.send('ok'));

    const agent = request(app);
    await agent.get('/test').set('x-api-key', 'k1').expect(200);
    await agent.get('/test').set('x-api-key', 'k1').expect(429);
    await agent.get('/test').set('x-api-key', 'k2').expect(200);
  });

  it('should skip if skip() returns true', async () => {
    app.get('/test', rateLimiter({
      windowMs: 1000,
      max: 1,
      skip: (req) => req.headers['x-no-limit'] === '1'
    }), (req, res) => res.send('ok'));

    const agent = request(app);
    await agent.get('/test').set('x-no-limit', '1').expect(200);
    await agent.get('/test').set('x-no-limit', '1').expect(200);
  });
});
```
