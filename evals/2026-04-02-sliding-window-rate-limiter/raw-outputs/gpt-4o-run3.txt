Creating a production-ready sliding window rate limiter for Express.js involves several steps. We need to ensure the code is robust, scalable, and easily maintainable. Below is a high-level implementation plan:

1. **Middleware Setup**: Create an Express middleware that can be added per route.
2. **In-Memory Storage**: Implement a storage mechanism using a JavaScript object.
3. **Redis Adapter**: Provide an optional Redis adapter for distributed environments.
4. **Headers and Responses**: Set appropriate response headers (`Retry-After`, `X-RateLimit-Limit`, `X-RateLimit-Remaining`, and `X-RateLimit-Reset`).
5. **TypeScript Types**: Define clear TypeScript interfaces for configurations.
6. **Unit Tests**: Ensure functionality with unit tests.

Here's a foundational implementation:

### Middleware Implementation

```typescript
import express, { Request, Response, NextFunction } from 'express';
import Redis from 'ioredis';

// TypeScript interfaces for rate limit configuration
interface RateLimitOptions {
  windowMs: number;
  max: number;
  message?: string;
}

interface RateLimiter {
  getKey(req: Request): string;
  checkLimit(key: string, options: RateLimitOptions): Promise<{ remaining: number, resetTime: number }>;
}

class InMemoryStore implements RateLimiter {
  private hits: Map<string, { count: number, timestamp: number }> = new Map();

  getKey(req: Request): string {
    return req.ip;
  }

  async checkLimit(key: string, options: RateLimitOptions): Promise<{ remaining: number, resetTime: number }> {
    const currentTime = Date.now();
    const resetTime = currentTime + options.windowMs;
    const entry = this.hits.get(key);

    if (!entry || currentTime > entry.timestamp + options.windowMs) {
      this.hits.set(key, { count: 1, timestamp: currentTime });
      return { remaining: options.max - 1, resetTime };
    }

    if (entry.count < options.max) {
      entry.count += 1;
      this.hits.set(key, entry);
      return { remaining: options.max - entry.count, resetTime: entry.timestamp + options.windowMs };
    }

    return { remaining: 0, resetTime: entry.timestamp + options.windowMs };
  }
}

class RedisStore implements RateLimiter {
  private client: Redis.Redis;

  constructor(redisClient: Redis.Redis) {
    this.client = redisClient;
  }

  getKey(req: Request): string {
    return req.ip;
  }

  async checkLimit(key: string, options: RateLimitOptions): Promise<{ remaining: number, resetTime: number }> {
    const currentTime = Date.now();
    const resetTime = currentTime + options.windowMs;
    const count = await this.client.incr(key);
    
    if (count === 1) {
      await this.client.pexpire(key, options.windowMs);
    }

    const ttl = await this.client.pttl(key);
    const remaining = Math.max(0, options.max - count);

    return { remaining, resetTime: currentTime + ttl };
  }
}

function rateLimiterMiddleware(options: RateLimitOptions, store: RateLimiter = new InMemoryStore()) {
  return async (req: Request, res: Response, next: NextFunction) => {
    const key = store.getKey(req);
    const { remaining, resetTime } = await store.checkLimit(key, options);

    res.setHeader('X-RateLimit-Limit', options.max);
    res.setHeader('X-RateLimit-Remaining', remaining);
    res.setHeader('X-RateLimit-Reset', resetTime);

    if (remaining <= 0) {
      res.setHeader('Retry-After', Math.ceil((resetTime - Date.now()) / 1000));
      return res.status(429).send(options.message || 'Too many requests, please try again later.');
    }

    next();
  };
}

// Example setup with Express.js
const app = express();

// In-memory rate limiting
app.use('/api/some-route', rateLimiterMiddleware({ windowMs: 60000, max: 100 }));

// Using Redis for distributed rate limiting
// const redisClient = new Redis();
// app.use('/api/some-other-route', rateLimiterMiddleware({ windowMs: 60000, max: 100 }, new RedisStore(redisClient)));

app.get('/api/some-route', (req, res) => {
  res.send('Hello, world!');
});

app.listen(3000, () => {
  console.log('Server is running on port 3000');
});
```

### Testing

For testing, you can use a tool like Jest to ensure your middleware functions as expected:

```typescript
import request from 'supertest';
import express from 'express';
// Import your middleware and interfaces here

describe('Rate Limiter Middleware', () => {
  let app: express.Application;

  beforeEach(() => {
    app = express();
    app.use('/api/some-route', rateLimiterMiddleware({ windowMs: 1000, max: 2 }));
    app.get('/api/some-route', (req, res) => {
      res.send('Hello, world!');
    });
  });

  it('should allow requests under the limit', async () => {
    const res1 = await request(app).get('/api/some-route');
    const res2 = await request(app).get('/api/some-route');

    expect(res1.status).toBe(200);
    expect(res2.status).toBe(200);
    expect(res2.header['x-ratelimit-remaining']).toBe('0');
  });

  it('should block requests over the limit', async () => {
    await request(app).get('/api/some-route');
    await request(app).get('/api/some-route');
    const res3 = await request(app).get('/api/some-route');

    expect(res3.status).toBe(429);
    expect(res3.header['retry-after']).toBeDefined();
  });
});
```

### Explanation

- **`InMemoryStore` and `RedisStore`**: Both stores conform to a common interface (`RateLimiter`), ensuring pluggability. `InMemoryStore` operates within a single process, while `RedisStore` is suitable for distributed deployments leveraging Redis.
- **Headers**: The middleware sets the necessary headers for rate limiting to inform clients of their current usage.
- **Tests**: Basic tests ensure the middleware respects both the request limits and the time windows.
- **Scalability**: This setup allows easy swapping between in-memory and Redis-backed storage, enabling scalable distributed deployments.

This foundation should help get you started in building a rate limiter for an Express.js application, with clear pathways for both local and distributed deployments.
