### `src/types.ts`

```typescript
import { Request, Response, NextFunction } from 'express';

export interface RateLimitInfo {
  limit: number;
  current: number;
  remaining: number;
  resetTime: Date;
}

export interface RateLimitOptions {
  windowMs: number;
  max: number;
  message?: string | ((req: Request, res: Response) => string);
  statusCode?: number;
  keyGenerator?: (req: Request) => string;
  skip?: (req: Request) => boolean | Promise<boolean>;
  handler?: (req: Request, res: Response, next: NextFunction) => void;
  onLimitReached?: (req: Request, res: Response, options: RateLimitInfo) => void;
  store?: Store;
  standardHeaders?: boolean;
  legacyHeaders?: boolean;
}

export interface Store {
  /**
   * Increment the hit count for a key
   * @param key - The identifier for the client
   * @returns Promise with rate limit info
   */
  increment(key: string): Promise<RateLimitInfo>;

  /**
   * Reset the hit count for a key
   * @param key - The identifier for the client
   */
  resetKey(key: string): Promise<void>;

  /**
   * Get the current rate limit info for a key
   * @param key - The identifier for the client
   */
  get(key: string): Promise<RateLimitInfo | undefined>;

  /**
   * Clean up expired entries
   */
  cleanup?(): Promise<void>;

  /**
   * Reset all keys (primarily for testing)
   */
  resetAll?(): Promise<void>;
}

export interface WindowEntry {
  count: number;
  resetTime: number;
  requests: number[];
}
```

### `src/stores/MemoryStore.ts`

```typescript
import { Store, RateLimitInfo, WindowEntry } from '../types';

export class MemoryStore implements Store {
  private store: Map<string, WindowEntry>;
  private windowMs: number;
  private max: number;
  private cleanupInterval?: NodeJS.Timeout;

  constructor(windowMs: number, max: number) {
    this.store = new Map();
    this.windowMs = windowMs;
    this.max = max;

    // Auto cleanup every minute
    this.startCleanup();
  }

  private startCleanup(): void {
    this.cleanupInterval = setInterval(() => {
      this.cleanup();
    }, 60000); // Run cleanup every minute

    // Don't prevent the process from exiting
    if (this.cleanupInterval.unref) {
      this.cleanupInterval.unref();
    }
  }

  async increment(key: string): Promise<RateLimitInfo> {
    const now = Date.now();
    const windowStart = now - this.windowMs;
    
    let entry = this.store.get(key);

    if (!entry || entry.resetTime <= now) {
      // Create new window
      entry = {
        count: 0,
        resetTime: now + this.windowMs,
        requests: []
      };
      this.store.set(key, entry);
    }

    // Remove requests outside the sliding window
    entry.requests = entry.requests.filter(timestamp => timestamp > windowStart);
    
    // Add current request
    entry.requests.push(now);
    entry.count = entry.requests.length;

    const resetTime = new Date(now + this.windowMs);
    const current = entry.count;
    const remaining = Math.max(0, this.max - current);

    return {
      limit: this.max,
      current,
      remaining,
      resetTime
    };
  }

  async resetKey(key: string): Promise<void> {
    this.store.delete(key);
  }

  async get(key: string): Promise<RateLimitInfo | undefined> {
    const now = Date.now();
    const windowStart = now - this.windowMs;
    const entry = this.store.get(key);

    if (!entry || entry.resetTime <= now) {
      return {
        limit: this.max,
        current: 0,
        remaining: this.max,
        resetTime: new Date(now + this.windowMs)
      };
    }

    // Filter requests in current window
    const validRequests = entry.requests.filter(timestamp => timestamp > windowStart);
    const current = validRequests.length;
    
    return {
      limit: this.max,
      current,
      remaining: Math.max(0, this.max - current),
      resetTime: new Date(entry.resetTime)
    };
  }

  async cleanup(): Promise<void> {
    const now = Date.now();
    const keysToDelete: string[] = [];

    for (const [key, entry] of this.store.entries()) {
      if (entry.resetTime <= now) {
        keysToDelete.push(key);
      }
    }

    keysToDelete.forEach(key => this.store.delete(key));
  }

  async resetAll(): Promise<void> {
    this.store.clear();
  }

  destroy(): void {
    if (this.cleanupInterval) {
      clearInterval(this.cleanupInterval);
    }
    this.store.clear();
  }
}
```
