```typescript
import { Request, Response, NextFunction } from 'express';

export interface RateLimitInfo {
  limit: number;
  current: number;
  remaining: number;
  resetTime: Date;
}

export interface RateLimitResult {
  allowed: boolean;
  limit: number;
  current: number;
  remaining: number;
  resetTime: number;
}

export interface RateLimiterOptions {
  /**
   * Time window in milliseconds
   * @default 60000 (1 minute)
   */
  windowMs?: number;

  /**
   * Maximum number of requests per window
   * @default 100
   */
  max?: number;

  /**
   * Storage adapter
   * @default MemoryStore
   */
  store?: Store;

  /**
   * Key generator function
   * @default IP address
   */
  keyGenerator?: (req: Request) => string;

  /**
   * Skip requests that match this function
   */
  skip?: (req: Request) => boolean | Promise<boolean>;

  /**
   * Handler called when rate limit is exceeded
   */
  handler?: (req: Request, res: Response, next: NextFunction) => void;

  /**
   * Customize response headers
   */
  standardHeaders?: boolean;

  /**
   * Add legacy X-RateLimit headers
   */
  legacyHeaders?: boolean;

  /**
   * Skip failed requests (requests that resulted in an error)
   */
  skipFailedRequests?: boolean;

  /**
   * Skip successful requests (requests that didn't result in an error)
   */
  skipSuccessfulRequests?: boolean;

  /**
   * Custom message when rate limit is exceeded
   */
  message?: string | ((req: Request, res: Response) => string);

  /**
   * Status code to send when rate limit is exceeded
   * @default 429
   */
  statusCode?: number;
}

export interface Store {
  /**
   * Increment the counter for a key
   */
  increment(key: string): Promise<RateLimitResult>;

  /**
   * Decrement the counter for a key
   */
  decrement(key: string): Promise<void>;

  /**
   * Reset the counter for a key
   */
  resetKey(key: string): Promise<void>;

  /**
   * Reset all counters
   */
  resetAll(): Promise<void>;

  /**
   * Get current info for a key
   */
  get(key: string): Promise<RateLimitInfo | undefined>;
}

export interface StoreOptions {
  windowMs: number;
  max: number;
}
```

### `src/stores/Store.ts`

```typescript
import { Store, StoreOptions, RateLimitInfo, RateLimitResult } from '../types';

export abstract class BaseStore implements Store {
  protected windowMs: number;
  protected max: number;

  constructor(options: StoreOptions) {
    this.windowMs = options.windowMs;
    this.max = options.max;
  }

  abstract increment(key: string): Promise<RateLimitResult>;
  abstract decrement(key: string): Promise<void>;
  abstract resetKey(key: string): Promise<void>;
  abstract resetAll(): Promise<void>;
  abstract get(key: string): Promise<RateLimitInfo | undefined>;

  protected calculateResetTime(): number {
    return Date.now() + this.windowMs;
  }

  protected isExpired(resetTime: number): boolean {
    return Date.now() >= resetTime;
  }
}
```

### `src/stores/MemoryStore.ts`

```typescript
import { BaseStore } from './Store';
import { RateLimitInfo, RateLimitResult, StoreOptions } from '../types';

interface WindowEntry {
  timestamps: number[];
  resetTime: number;
}

export class MemoryStore extends BaseStore {
  private hits: Map<string, WindowEntry>;
  private cleanupInterval: NodeJS.Timeout;
  private readonly cleanupIntervalMs = 60000; // Clean up every minute

  constructor(options: StoreOptions) {
    super(options);
    this.hits = new Map();
    
    // Periodic cleanup to prevent memory leaks
    this.cleanupInterval = setInterval(() => {
      this.cleanup();
    }, this.cleanupIntervalMs);
    
    // Don't block Node.js from exiting
    if (this.cleanupInterval.unref) {
      this.cleanupInterval.unref();
    }
  }

  async increment(key: string): Promise<RateLimitResult> {
    const now = Date.now();
    const windowStart = now - this.windowMs;
    
    let entry = this.hits.get(key);
    
    if (!entry || this.isExpired(entry.resetTime)) {
      // Create new entry
      entry = {
        timestamps: [now],
        resetTime: this.calculateResetTime(),
      };
      this.hits.set(key, entry);
    } else {
      // Filter out timestamps outside the sliding window
      entry.timestamps = entry.timestamps.filter(ts => ts > windowStart);
      entry.timestamps.push(now);
    }

    const current = entry.timestamps.length;
    const remaining = Math.max(0, this.max - current);
    const allowed = current <= this.max;

    return {
      allowed,
      limit: this.max,
      current,
      remaining,
      resetTime: entry.resetTime,
    };
  }

  async decrement(key: string): Promise<void> {
    const entry = this.hits.get(key);
    if (entry && entry.timestamps.length > 0) {
      entry.timestamps.pop();
    }
  }

  async resetKey(key: string): Promise<void> {
    this.hits.delete(key);
  }

  async resetAll(): Promise<void> {
    this.hits.clear();
  }

  async get(key: string): Promise<RateLimitInfo | undefined> {
    const entry = this.hits.get(key);
    if (!entry || this.isExpired(entry.resetTime)) {
      return undefined;
    }

    const now = Date.now();
    const windowStart = now - this.windowMs;
    const validTimestamps = entry.timestamps.filter(ts => ts > windowStart);
    const current = validTimestamps.length;

    return {
      limit: this.max,
      current,
      remaining: Math.max(0, this.max - current),
      resetTime: new Date(entry.resetTime),
    };
  }

  private cleanup(): void {
    const now = Date.now();
    const keysToDelete: string[] = [];

    for (const [key, entry] of this.hits.entries()) {
      if (this.isExpired(entry.resetTime)) {
        keysToDelete.push(key);
      } else {
        // Clean up old timestamps within valid entries
        const windowStart = now - this.windowMs;
        entry.timestamps = entry.timestamps.filter(ts => ts > windowStart);
        
        if (entry.timestamps.length === 0) {
          keysToDelete.push(key);
        }
      }
    }

    for (const key of keysToDelete) {
      this.hits.delete(key);
    }
  }

  destroy(): void {
    clearInterval(this.cleanupInterval);
    this.hits.clear();
  }
}
```
