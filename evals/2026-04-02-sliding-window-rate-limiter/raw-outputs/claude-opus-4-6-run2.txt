```typescript
import { Request, Response, NextFunction } from "express";

/**
 * Represents a single request record within the sliding window.
 */
export interface RequestRecord {
  timestamp: number;
}

/**
 * Result of a rate limit check against the store.
 */
export interface RateLimitResult {
  /** Whether the request is allowed */
  allowed: boolean;
  /** Total number of allowed requests in the window */
  limit: number;
  /** Number of remaining requests in the current window */
  remaining: number;
  /** Unix timestamp (seconds) when the oldest request in the window expires */
  resetAt: number;
  /** Seconds until the client should retry (only set when blocked) */
  retryAfter?: number;
}

/**
 * Configuration for the rate limiter middleware.
 */
export interface RateLimiterOptions {
  /** Maximum number of requests allowed within the window. Default: 100 */
  maxRequests?: number;

  /** Sliding window size in milliseconds. Default: 60000 (1 minute) */
  windowMs?: number;

  /**
   * Function to derive the rate-limit key from the request.
   * Default: uses `req.ip`.
   */
  keyGenerator?: (req: Request) => string;

  /**
   * Pluggable storage backend. Default: in-memory store.
   */
  store?: RateLimitStore;

  /**
   * Custom handler invoked when a request is rate-limited.
   * If not provided, a default 429 JSON response is sent.
   */
  handler?: (
    req: Request,
    res: Response,
    next: NextFunction,
    result: RateLimitResult
  ) => void;

  /**
   * Whether to include standard rate limit headers. Default: true
   */
  headers?: boolean;

  /**
   * Whether to skip failed requests (status >= 400). Default: false
   */
  skipFailedRequests?: boolean;

  /**
   * Whether to skip successful requests (status < 400). Default: false
   */
  skipSuccessfulRequests?: boolean;

  /**
   * Optional function to skip rate limiting for certain requests entirely.
   */
  skip?: (req: Request) => boolean | Promise<boolean>;

  /**
   * Prefix for store keys, useful when sharing a Redis instance. Default: 'rl:'
   */
  keyPrefix?: string;
}

/**
 * Interface that all storage backends must implement.
 */
export interface RateLimitStore {
  /**
   * Record a hit and return the current rate limit state.
   * @param key - The unique identifier for the client+route
   * @param windowMs - The sliding window duration in ms
   * @param maxRequests - The maximum number of requests in the window
   */
  hit(
    key: string,
    windowMs: number,
    maxRequests: number
  ): Promise<RateLimitResult>;

  /**
   * Decrement the counter for the key (used when skipping requests).
   */
  decrement(key: string): Promise<void>;

  /**
   * Reset the counter for a specific key.
   */
  reset(key: string): Promise<void>;

  /**
   * Clean up resources (timers, connections, etc.).
   */
  destroy?(): Promise<void>;
}

/**
 * Augment Express Request to expose rate limit info downstream.
 */
declare global {
  namespace Express {
    interface Request {
      rateLimit?: RateLimitResult;
    }
  }
}
```

### `src/stores/memory.store.ts`

```typescript
import { RateLimitStore, RateLimitResult } from "../types";

interface WindowEntry {
  timestamps: number[];
}

/**
 * In-memory sliding window rate limit store.
 *
 * Uses a sorted array of timestamps per key. On each hit, expired entries
 * outside the window are pruned. A periodic sweep removes stale keys entirely.
 */
export class MemoryStore implements RateLimitStore {
  private store: Map<string, WindowEntry> = new Map();
  private cleanupTimer: ReturnType<typeof setInterval> | null = null;
  private readonly cleanupIntervalMs: number;

  constructor(options?: { cleanupIntervalMs?: number }) {
    this.cleanupIntervalMs = options?.cleanupIntervalMs ?? 60_000;
    this.startCleanup();
  }

  async hit(
    key: string,
    windowMs: number,
    maxRequests: number
  ): Promise<RateLimitResult> {
    const now = Date.now();
    const windowStart = now - windowMs;

    let entry = this.store.get(key);
    if (!entry) {
      entry = { timestamps: [] };
      this.store.set(key, entry);
    }

    // Prune timestamps outside the current window
    entry.timestamps = entry.timestamps.filter((t) => t > windowStart);

    // Calculate current count BEFORE adding the new request
    const currentCount = entry.timestamps.length;
    const allowed = currentCount < maxRequests;

    if (allowed) {
      entry.timestamps.push(now);
    }

    const remaining = allowed
      ? maxRequests - currentCount - 1
      : 0;

    // resetAt: when the oldest timestamp in the window will expire
    const oldestInWindow =
      entry.timestamps.length > 0 ? entry.timestamps[0] : now;
    const resetAt = Math.ceil((oldestInWindow + windowMs) / 1000);

    const result: RateLimitResult = {
      allowed,
      limit: maxRequests,
      remaining: Math.max(0, remaining),
      resetAt,
    };

    if (!allowed) {
      // Time until the oldest request falls out of the window
      const retryAfterMs = oldestInWindow + windowMs - now;
      result.retryAfter = Math.ceil(retryAfterMs / 1000);
    }

    return result;
  }

  async decrement(key: string): Promise<void> {
    const entry = this.store.get(key);
    if (entry && entry.timestamps.length > 0) {
      // Remove the most recent timestamp
      entry.timestamps.pop();
    }
  }

  async reset(key: string): Promise<void> {
    this.store.delete(key);
  }

  async destroy(): Promise<void> {
    if (this.cleanupTimer) {
      clearInterval(this.cleanupTimer);
      this.cleanupTimer = null;
    }
    this.store.clear();
  }

  /** Visible for testing */
  get size(): number {
    return this.store.size;
  }

  private startCleanup(): void {
    this.cleanupTimer = setInterval(() => {
      const now = Date.now();
      // We don't know per-key windowMs, so we use a generous 5-minute TTL
      // for the sweep. Actual per-request pruning happens in `hit()`.
      for (const [key, entry] of this.store.entries()) {
        // Remove keys with no timestamps or all timestamps older than 5 min
        if (
          entry.timestamps.length === 0 ||
          entry.timestamps.every((t) => t < now - 300_000)
        ) {
          this.store.delete(key);
        }
      }
    }, this.cleanupIntervalMs);

    // Allow the Node process to exit even if the timer is active
    if (this.cleanupTimer?.unref) {
      this.cleanupTimer.unref();
    }
  }
}
```

/**
 * Lua script for atomic sliding window rate limiting.
 *
 * KEYS[1] = the sorted set key
 * ARGV[1] = window start timestamp (ms)
 * ARGV[2] = current timestamp (ms) — used as both score and member uniqueness
 * ARGV[3] = max requests
 * ARGV[4] = window size (ms) for pexpire
 * ARGV[5] = unique member suffix (to avoid collisions at same ms)
 *
 * Returns: { count_before_add, oldest_score, was_added (0/1) }
 */
const SLIDING_WINDOW_LUA = `
local key = KEYS[1]
local windowStart = tonumber(ARGV[1])
local now = tonumber(ARGV[2])
local maxRequests = tonumber(ARGV[3])
local windowMs = tonumber(ARGV[4])
local uniqueSuffix = ARGV[5]

-- Remove expired entries
redis.call('ZREMRANGEBYSCORE', key, '-inf', windowStart)

-- Count current entries
local currentCount = redis.call('ZCARD', key)

local added = 0
if currentCount < maxRequests then
  -- Add new entry with score = now, member = now:uniqueSuffix
  redis.call('ZADD', key, now, now .. ':' .. uniqueSuffix)
  added = 1
end

-- Get oldest entry score
local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
local oldestScore = 0
if #oldest >= 2 then
  oldestScore = tonumber(oldest[2])
end

-- Set expiry on the key
redis.call('PEXPIRE', key, windowMs)

return {currentCount, oldestScore, added}
`;
