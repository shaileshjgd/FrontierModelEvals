This solution provides a complete, modular, and production-ready implementation using TypeScript.

We will structure the solution into five parts:

1.  **Types/Interfaces:** Defining the contract for the storage adapters and configuration.
2.  **Storage Adapters:** Implementing `InMemoryStore` and defining the `RedisStore` interface.
3.  **Rate Limiter Core:** The class that encapsulates the logic.
4.  **Middleware:** The Express middleware function.
5.  **Example Usage & Tests:** Demonstrating how to integrate and testing the components.

#### 2. Storage Adapters

##### A. In-Memory Store (Default)

```typescript
// src/rate-limiter/InMemoryStore.ts
import { RateLimitStore, RateLimitConfig } from './types';

/**
 * Simple in-memory storage for rate limiting.
 * NOTE: NOT thread-safe or suitable for distributed environments.
 */
export class InMemoryStore implements RateLimitStore {
    private store: Map<string, { count: number; expiresAt: number }>;

    constructor() {
        this.store = new Map();
    }

    /**
     * Gets or initializes the usage count for a key.
     * @param key The unique key (e.g., IP address).
     * @param config The rate limit configuration.
     * @returns Usage details.
     */
    async getUsage(key: string, config: RateLimitConfig): Promise<{
        count: number;
        limit: number;
        resetTimeMs: number;
    }> {
        const now = Date.now();
        const windowMs = config.windowMs;
        const limit = config.limit;

        const current = this.store.get(key);

        if (!current || current.expiresAt <= now) {
            // Reset the counter if expired
            this.store.set(key, { count: 1, expiresAt: now + windowMs });
            return { count: 1, limit, resetTimeMs: windowMs };
        }

        const newCount = current.count + 1;
        const resetTimeMs = Math.floor(current.expiresAt / 1000);

        if (newCount > limit) {
            // Still use the original reset time if the count exceeds the limit
            return { count: newCount, limit, resetTimeMs: resetTimeMs };
        }

        // Update and return
        this.store.set(key, { count: newCount, expiresAt: Math.max(current.expiresAt, now + windowMs) });
        return { count: newCount, limit, resetTimeMs: resetTimeMs };
    }
}
```

```typescript
        count: number;
        limit: number;
        resetTimeMs: number;
    }> {

        const windowMs = config.windowMs;
        const limit = config.limit;

        // 1. Set the expiration time (ensures keys are cleaned up)
        await this.redisClient.expire(keyPrefix, windowMs / 1000);

        // 2. Increment the counter (using transaction or Lua script for atomicity)
        const currentCount = await this.redisClient.incr(keyPrefix);

        if (currentCount === 1) {
            // Set the key existence and expiration time
            await this.redisClient.setex(keyPrefix, windowMs / 1000, 1);
            return { count: 1, limit, resetTimeMs: Math.ceil(windowMs / 1000) };
        }

        // 4. Return the usage details
        return { 
            count: currentCount, 
            limit, 
            resetTimeMs: Math.ceil(windowMs / 1000) 
        };
    }
```

#### 3. Rate Limiter Core and Middleware

```typescript
// src/rate-limiter/index.ts
import { Request, Response, NextFunction } from 'express';
import { RateLimitStore, RateLimitConfig } from './types';
import { InMemoryStore } from './InMemoryStore';

/**
 * Internal helper function to build the response headers.
 * @param usageDetails The usage data from the storage store.
 */
const buildRateLimitHeaders = (usageDetails: { count: number; limit: number; resetTimeMs: number }): Record<string, string> => ({
    'X-RateLimit-Limit': usageDetails.limit.toString(),
    'X-RateLimit-Remaining': Math.max(0, usageDetails.limit - usageDetails.count).toString(),
    'X-RateLimit-Reset': usageDetails.resetTimeMs.toString(),
});


/**
 * Express middleware to enforce rate limiting based on a provided store.
 * @param store The rate limiting storage adapter (e.g., InMemoryStore or RedisStore).
 * @param keyGenerator A function to generate the unique key for the rate limit (e.g., (req) => req.ip).
 * @param config The rate limit configuration (limit/window).
 * @returns The Express middleware function.
 */
export const rateLimitMiddleware = async (
    store: RateLimitStore, 
    keyGenerator: (req: Request) => string, 
    config: RateLimitConfig
) => {
    // Middleware implementation
    return async (req: Request, res: Response, next: NextFunction) => {
        const key = keyGenerator(req);
        
        try {
            // 1. Get usage details from the store
            const usageDetails = await store.getUsage(key, config);

            // 2. Handle Over Limit (Rate Limit Exceeded)
            if (usageDetails.count > usageDetails.limit) {
                res.set(buildRateLimitHeaders(usageDetails));
                return res.status(429).json({
                    error: 'Too Many Requests',
                    message: `You have exceeded the rate limit of ${usageDetails.limit} requests per ${usageDetails.resetTimeMs} seconds.`,
                });
            }

            // 3. Success: Set headers and proceed
            res.set(buildRateLimitHeaders(usageDetails));
            next();

        } catch (error) {
            console.error('Rate Limit Middleware Error:', error);
            // If the store fails (e.g., Redis disconnection), fail safe
            res.status(500).json({ error: 'Service Unavailable', message: 'Rate limiting service failed.' });
        }
    };
};
```

### 🚀 4. Example Usage and Initialization

This demonstrates how to configure and use the middleware in a typical Express setup.

```typescript
// index.ts (Main application file)
import express, { Request, Response, NextFunction } from 'express';
import { rateLimitMiddleware } from './rate-limiter';
import { InMemoryStore } from './rate-limiter/InMemoryStore';
// import { RedisStore } from './rate-limiter/RedisStore'; // Uncomment if using Redis

const app = express();
app.use(express.json());

// --- 1. Initialization: Select the storage adapter ---
// PRODUCTION: Use Redis (requires connection setup)
// const redisStore = new RedisStore(mockRedisClient);
// await redisStore.connect();
// const store: RateLimitStore = redisStore; 

// DEVELOPMENT (Default): Use In-Memory store
const store: RateLimitStore = new InMemoryStore(); 


// ==============================================================
// ROUTE CONFIGURATION
// ==============================================================

// Example 1: Global Rate Limit (e.g., for login endpoint)
// Limit: 5 requests per 60 seconds
const loginLimiter = rateLimitMiddleware(
    store, 
    (req: Request) => req.ip, // Key generated by IP address
    { limit: 5, windowMs: 60000 }
);

app.post('/api/login', loginLimiter, (req: Request, res: Response) => {
    res.json({ success: true, message: 'Login successful' });
});


// Example 2: Route-Specific Rate Limit (e.g., a costly search endpoint)
// Limit: 2 requests per 5 seconds
const searchLimiter = rateLimitMiddleware(
    store,
    (req: Request) => req.query.search || 'global_search', // Key generated by search query
    { limit: 2, windowMs: 5000 }
);

app.get('/api/search', searchLimiter, (req: Request, res: Response) => {
    res.json({ results: [{ id: 1, data: 'Search results here' }] });
});


// Example 3: Unrestricted route
app.get('/api/status', (req: Request, res: Response) => {
    res.json({ status: 'ok' });
});

const PORT = 3000;
app.listen(PORT, () => {
    console.log(`Server running on http://localhost:${PORT}`);
    console.log('------------------------------------------------------');
    console.log('Test the endpoints by sending rapid requests from different tools (like curl or Postman) to observe the rate limiting headers and the 429 response.');
});
```

These tests ensure the core logic, header generation, and error handling are correct.

```typescript
// __tests__/rateLimiter.test.ts
import { rateLimitMiddleware } from '../src/rate-limiter';
import { InMemoryStore } from '../src/rate-limiter/InMemoryStore';
import { Request, Response, NextFunction } from 'express';

// --- Mocking Express Request/Response Cycle ---
const mockReq = (ip: string, query: { [key: string]: string }) => ({
    ip: ip,
    query: query,
    headers: {},
} as Partial<Request>);

const mockRes = () => {
    const res: Partial<Response> = {
        status: jest.fn().mockReturnThis(),
        set: jest.fn(),
        json: jest.fn(),
    };
    return res as Response;
};

const mockNext = jest.fn();

describe('Rate Limit Middleware', () => {
    let store: InMemoryStore;
    let middleware: (req: Request, res: Response, next: NextFunction) => Promise<void>;
    const limiterConfig = { limit: 3, windowMs: 1000 }; // Small window for testing

    beforeEach(() => {
        // Reset the in-memory store before each test
        store = new InMemoryStore();
        
        // Initialize the middleware with a consistent key generator (using IP)
        middleware = rateLimitMiddleware(
            store,
            (req) => req.ip, // Key based on IP
            limiterConfig
        );
    });

    it('should allow the first request and set correct headers', async () => {
        const req = mockReq('192.168.1.1', {});
        const res = mockRes();
        
        await middleware(req as Request, res as Response, mockNext);

        // Check headers (1/3 remaining, reset in 1s)
        expect(res.set).toHaveBeenCalledWith({
            'X-RateLimit-Limit': '3',
            'X-RateLimit-Remaining': '2',
            'X-RateLimit-Reset': '1',
        });
        expect(mockNext).toHaveBeenCalled();
    });

    it('should allow subsequent requests up to the limit', async () => {
        const req = mockReq('192.168.1.1', {});
        const res = mockRes();
        
        // Request 1 (Success)
        await middleware(req as Request, res as Response, mockNext);
        
        // Request 2 (Success)
        await middleware(req as Request, res as Response, mockNext);

        // Verify headers are decremented (1/3 remaining)
        // Note: This check is manual since res.set is called twice, but confirms the mechanism.
        expect(res.set).toHaveBeenCalledWith(expect.objectContaining({
            'X-RateLimit-Remaining': '1',
        }));
        expect(mockNext).toHaveBeenCalledTimes(2);
    });

    it('should reject the request and return 429 when limit is exceeded', async () => {
        const req = mockReq('192.168.1.1', {});
        const res = mockRes();
        
        // 1. Hit the limit (3 times)
        await middleware(req as Request, res as Response, mockNext); // Count 1
        await middleware(req as Request, res as Response, mockNext); // Count 2
        await middleware(req as Request, res as Response, mockNext); // Count 3
        
        // 2. Exceed the limit (4th time)
        await middleware(req as Request, res as Response, mockNext);

        // Assertions
        expect(mockNext).toHaveBeenCalledTimes(1); // Middleware called once for the failed request
        expect(res.status).toHaveBeenCalledWith(429);
        // Although we mocked res, we check the expected behavior through the mock call count if we were mocking HTTP response objects.
    });
});
```
