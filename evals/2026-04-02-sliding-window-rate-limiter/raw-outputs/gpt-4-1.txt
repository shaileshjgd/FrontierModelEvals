## 🏗️ Strategy & Architecture

### Core Parts

1. **Sliding Window Algorithm**  
   - Not just a fixed window. We'll maintain a timestamped list of request events, expiring old entries outside the sliding window.
   - The number of requests within `[now - windowMs, now]` must not exceed the limit.

2. **Per-route Configuration**  
   - Allow for a configuration object (e.g., `limits: { "/api/login": { windowMs, max }, ... }`) so each route can have its own limit.

3. **Storage Abstraction**  
   - **In-memory**: Default. For each key (e.g., user IP + route), record timestamps of hits.
   - **Redis Adapter**: Replaceable strategy to persist data, for distributed deployments.

4. **Headers and Responses**  
   - On 429, return `Retry-After`, `X-RateLimit-Limit`, `X-RateLimit-Remaining`, and `X-RateLimit-Reset` headers.

5. **Typescript**  
   - Use proper types/interfaces throughout.

6. **Testing**  
   - Unit tests for the core rate-limiting algorithm (sliding window).
   - Unit tests for the middleware.

## 2. **Storage Layer**

### **In-Memory Store**

```ts
// rateLimiter/storage/MemoryStore.ts
import { SlidingWindowStore } from '../types';

interface Hit {
  timestamp: number;
}

export class MemoryStore implements SlidingWindowStore {
  private hits: Map<string, Hit[]> = new Map();

  async incr(key: string, windowMs: number, timestamp: number) {
    let keyHits = this.hits.get(key) || [];
    // Remove old entries out of window
    keyHits = keyHits.filter(hit => hit.timestamp > (timestamp - windowMs));
    keyHits.push({ timestamp });

    this.hits.set(key, keyHits);

    const count = keyHits.length;
    // Find when the oldest request in the window will fall out = reset time
    const reset = keyHits[0]?.timestamp + windowMs || (timestamp + windowMs);
    return { count, reset };
  }

  async get(key: string, windowMs: number, timestamp: number) {
    let keyHits = this.hits.get(key) || [];
    keyHits = keyHits.filter(hit => hit.timestamp > (timestamp - windowMs));
    if (keyHits.length === 0) {
      this.hits.delete(key);
      return { count: 0, reset: timestamp + windowMs };
    }
    // Find when the oldest will fall out
    const reset = keyHits[0].timestamp + windowMs;
    return { count: keyHits.length, reset };
  }

  async resetKey(key: string) {
    this.hits.delete(key);
  }
}
```

### **Redis Store**

```ts
// rateLimiter/storage/RedisStore.ts
import { RedisClientType } from 'redis';
import { SlidingWindowStore } from '../types';

export class RedisStore implements SlidingWindowStore {
  private redis: RedisClientType;

  constructor(redis: RedisClientType) {
    this.redis = redis;
  }

  private keyName(key: string): string {
    return `rate-limiter:${key}`;
  }

  async incr(key: string, windowMs: number, timestamp: number) {
    const redisKey = this.keyName(key);
    // Use Redis sorted set to store timestamps
    const now = timestamp;
    const minTime = now - windowMs;
    const pipe = this.redis.multi();

    pipe.zRemRangeByScore(redisKey, 0, minTime); // remove old entries
    pipe.zAdd(redisKey, [{ score: now, value: '' + now }]);
    pipe.zCard(redisKey);
    pipe.zRange(redisKey, 0, 0); // get oldest in window
    pipe.pexpire(redisKey, windowMs);

    const [, , countRaw, oldestArr, ]: any = await pipe.exec();
    const count = Number(countRaw);
    const oldest = oldestArr?.[0] ? Number(oldestArr[0]) : now;
    const reset = oldest + windowMs;
    return { count, reset };
  }

  async get(key: string, windowMs: number, timestamp: number) {
    const redisKey = this.keyName(key);
    const now = timestamp;
    const minTime = now - windowMs;
    const pipe = this.redis.multi();

    pipe.zRemRangeByScore(redisKey, 0, minTime);
    pipe.zCard(redisKey);
    pipe.zRange(redisKey, 0, 0);

    const [, countRaw, oldestArr]: any = await pipe.exec();
    const count = Number(countRaw);
    const oldest = oldestArr?.[0] ? Number(oldestArr[0]) : now;
    const reset = count > 0 ? oldest + windowMs : now + windowMs;
    return { count, reset };
  }

  async resetKey(key: string) {
    await this.redis.del(this.keyName(key));
  }
}
```

## 3. **Sliding Window Algorithm Helper**

```ts
// rateLimiter/slidingWindow.ts
import { SlidingWindowStore } from './types';

export class SlidingWindowRateLimiter {
  private store: SlidingWindowStore;

  constructor(store: SlidingWindowStore) {
    this.store = store;
  }

  /**
   * Increments the count, returns info on allow/deny
   */
  async checkAndIncr(
    key: string,
    windowMs: number,
    max: number,
    timestamp = Date.now()
  ) {
    const { count, reset } = await this.store.incr(key, windowMs, timestamp);
    return {
      allowed: count <= max,
      count,
      reset,
    };
  }

  async getCurrent(
    key: string,
    windowMs: number,
    max: number,
    timestamp = Date.now()
  ) {
    const { count, reset } = await this.store.get(key, windowMs, timestamp);
    return {
      allowed: count < max,
      count,
      reset,
    };
  }

  async resetKey(key: string) {
    return this.store.resetKey(key);
  }
}
```

## 4. **Express Middleware Factory**

```ts
// rateLimiter/index.ts
import { Request, Response, NextFunction } from 'express';
import { SlidingWindowStore, RouteRateLimits, RateLimitConfig, KeyGenerator } from './types';
import { SlidingWindowRateLimiter } from './slidingWindow';
import { MemoryStore } from './storage/MemoryStore';

interface RateLimiterOptions {
  limits: RouteRateLimits;
  store?: SlidingWindowStore;
  keyGenerator?: KeyGenerator;
}

export function rateLimiter(options: RateLimiterOptions) {
  const store = options.store || new MemoryStore();
  const limiter = new SlidingWindowRateLimiter(store);
  const limits = options.limits;

  // Use IP as default key
  const keyGenerator: KeyGenerator = options.keyGenerator || ((req) =>
    req.ip || req.headers['x-forwarded-for']?.toString() || req.connection.remoteAddress || 'anon'
  );

  // Normalize route matches (basic - assumes exact paths)
  return async function rateLimitMiddleware(req: Request, res: Response, next: NextFunction) {
    const route = req.route ? req.route.path : req.path; // Express 4.x
    const config: RateLimitConfig | undefined = limits[route];

    if (!config) {
      return next();
    }

    const key = `${route}:${keyGenerator(req)}`;
    const timestamp = Date.now();
    const { allowed, count, reset } = await limiter.checkAndIncr(key, config.windowMs, config.max, timestamp);

    // Headers required by spec
    res.setHeader('X-RateLimit-Limit', config.max);
    res.setHeader('X-RateLimit-Remaining', Math.max(0, config.max - count));
    res.setHeader('X-RateLimit-Reset', Math.ceil(reset / 1000)); // seconds since epoch

    if (!allowed) {
      const retryAfter = Math.ceil((reset - timestamp) / 1000);
      res.setHeader('Retry-After', retryAfter);
      return res.status(429).json({
        message: 'Too many requests, please try again later',
      });
    }

    next();
  };
}
```

### **Testing the Sliding Window Logic**

```ts
// rateLimiter/__tests__/slidingWindow.test.ts
import { MemoryStore } from '../storage/MemoryStore';
import { SlidingWindowRateLimiter } from '../slidingWindow';

describe('SlidingWindowRateLimiter', () => {
  const windowMs = 60_000; // 1 min
  const max = 3;
  let limiter: SlidingWindowRateLimiter;

  beforeEach(() => {
    limiter = new SlidingWindowRateLimiter(new MemoryStore());
  });

  test('allows up to the limit within window', async () => {
    const now = Date.now();
    for (let i = 0; i < max; i++) {
      const result = await limiter.checkAndIncr('foo', windowMs, max, now);
      expect(result.allowed).toBe(true);
    }
    const result = await limiter.checkAndIncr('foo', windowMs, max, now);
    expect(result.allowed).toBe(false);
  });

  test('old requests are expired (sliding window)', async () => {
    const now = Date.now();
    // 3 requests within window
    for (let i = 0; i < max; i++) {
      expect((await limiter.checkAndIncr('foo', windowMs, max, now + i * 1000)).allowed).toBe(true);
    }
    // Next one is rejected
    expect((await limiter.checkAndIncr('foo', windowMs, max, now + 2000)).allowed).toBe(false);

    // Advance time so that first request is out of window
    const afterWindow = now + windowMs + 100;
    expect((await limiter.checkAndIncr('foo', windowMs, max, afterWindow)).allowed).toBe(true);
  });
});
```

```ts
// rateLimiter/__tests__/middleware.test.ts
import request from 'supertest';
import express from 'express';
import { rateLimiter } from '../index';
import { MemoryStore } from '../storage/MemoryStore';

describe('rateLimiter middleware', () => {
  let app: express.Express;
  beforeEach(() => {
    app = express();
    app.use(rateLimiter({
      limits: {
        '/login': { windowMs: 60_000, max: 2 },
      },
      store: new MemoryStore(),
      keyGenerator: (req) => req.headers['x-client-id'] as string || req.ip,
    }));
    app.get('/login', (req, res) => res.send('ok'));
    app.get('/other', (req, res) => res.send('ok'));
  });

  test('enforces rate limit per route', async () => {
    const agent = request(app);
    const headers = { 'x-client-id': 'abc' };
    expect((await agent.get('/login').set(headers)).status).toBe(200);
    expect((await agent.get('/login').set(headers)).status).toBe(200);
    const resp = await agent.get('/login').set(headers);
    expect(resp.status).toBe(429);
    expect(resp.headers['retry-after']).toBeDefined();
    expect(resp.headers['x-ratelimit-limit']).toBe('2');
    expect(resp.headers['x-ratelimit-remaining']).toBe('0');
    expect(resp.headers['x-ratelimit-reset']).toBeDefined();
    expect(resp.body.message).toMatch(/Too many requests/);
  });

  test('does not limit unconfigured route', async () => {
    const agent = request(app);
    expect((await agent.get('/other').set('x-client-id', 'abc')).status).toBe(200);
  });
});
```
