# Rate Limiter Middleware Architecture

## Strategy Overview

I'll build a production-ready rate limiter with the following components:

1. **Storage Abstraction Layer** - Interface for different storage backends
2. **Rate Limiting Algorithm** - Sliding window implementation
3. **Express.js Middleware** - Configurable rate limiting middleware
4. **Storage Implementations** - In-memory and Redis adapters
5. **TypeScript Types** - Strong typing throughout
6. **Comprehensive Testing** - Unit tests for all components

### 3. In-Memory Storage Adapter

```typescript
// storage/memory.ts
import { StorageAdapter } from './adapter';

export class InMemoryStorageAdapter implements StorageAdapter {
  private requests: Map<string, { 
    timestamps: number[]; 
    lastRequest: number;
  }> = new Map();

  async increment(key: string, windowMs: number): Promise<{ count: number; resetTime: number }> {
    const now = Date.now();
    if (!this.requests.has(key)) {
      this.requests.set(key, { timestamps: [], lastRequest: now });
    }

    const requestInfo = this.requests.get(key)!;
    const windowSize = windowMs;
    const windowStart = now - windowSize;

    // Clean old requests outside the window
    requestInfo.timestamps = requestInfo.timestamps.filter(
      timestamp => timestamp > windowStart
    );

    // Add current request
    requestInfo.timestamps.push(now);

    const count = requestInfo.timestamps.length;
    const resetTime = requestInfo.timestamps[0] || now;

    return {
      count,
      resetTime
    };
  }

  async resetKey(key: string): Promise<void> {
    this.requests.delete(key);
  }
}
```

### 4. Redis Storage Adapter

```typescript
// storage/redis.ts
import { createClient, RedisClientType } from 'redis';
import { StorageAdapter } from './adapter';

export class RedisStorageAdapter implements StorageAdapter {
  private client: RedisClientType;
  private windowKeys: Map<string, number> = new Map();

  constructor(redisUrl: string) {
    this.client = createClient({ url: redisUrl });
    this.client.connect();
  }

  async increment(key: string, windowMs: number): Promise<{ count: number; resetTime: number }> {
    const now = Date.now();
    const windowSize = windowMs;
    const keyBase = `rate_limit:${key}`;
    
    // Get recent requests within the window
    const windowStart = now - windowSize;
    const keyScores: [string, number][] = await this.client.ZRANGEBYSCORE(keyBase, windowStart, now, 'BYSCORE');
    
    // Add current request
    await this.client.ZADD(keyBase, { score: now, member: now.toString() });
    
    // Set expiration for cleanup
    await this.client.EXPIRE(keyBase, windowMs / 1000);
    
    return {
      count: keyScores.length + 1,
      resetTime: now + windowMs
    };
  }

  async resetKey(key: string): Promise<void> {
    const keyBase = `rate_limit:${key}`;
    await this.client.DEL(keyBase);
  }
}
```

### 5. Rate Limiter Core Implementation

```typescript
// rate-limiter.ts
import { StorageAdapter } from './adapter';

export class RateLimiter {
  private storage: StorageAdapter;
  private limits: Map<string, { windowMs: number; limit: number }>;

  constructor(storage: StorageAdapter, limits: Map<string, { windowMs: number; limit: number }>) {
    this.storage = storage;
    this.limits = limits;
  }

  async handleRequest(key: string, windowMs: number): Promise<boolean> {
    const { count, resetTime } = await this.storage.increment(key, windowMs);
    return count <= this.limits.get(key)?.limit;
  }
}
```

export const rateLimiterMiddleware = (rateLimiter: RateLimiter) => {
  return async (req: Request, res: Response, next: NextFunction) => {

    const config = getRouteConfig(req.path); // implement based on your route configuration
    const { count, resetTime } = await rateLimiter.handleRequest(key, config.windowMs);
    
    if (count > config.limit) {
      res.status(429).set({
        'Retry-After': Math.ceil((resetTime - Date.now()) / 1000),
        'X-RateLimit-Limit': config.limit,
        'X-RateLimit-Remaining': config.limit - count,
        'X-RateLimit-Reset': new Date(resetTime).toISOString()
      });
      return res.send('Rate limit exceeded');
    }
    next();
  };
};

### 7. Complete Implementation

```typescript
// index.ts
import { Request, Response, NextFunction } from 'express';

// RateLimiter class implementation
class RateLimiter {
  private limits: Map<string, number> = new Map();
  private storage: StorageAdapter;

  constructor(storage: StorageAdapter) {
    this.storage = storage;
  }

  addLimit(path: string, limit: number) {
    this.limits.set(path, limit);
  }

  async handleRequest(req: Request, res: Response, next: NextFunction) {
    const key = req.ip;
    const path = req.path;
    const limit = this.limits.get(path);
    
    if (!limit) {
      return next();
    }

    const count = await this.storage.increment(key, 60000); // 60 seconds window
    if (count > limit) {
      res.status(429).send('Too Many Requests');
      return;
    }

    next();
  }
}

// Export the rate limiter middleware
export { rateLimiterMiddleware, RateLimiter };
```

This implementation provides:

1. **Sliding Window Algorithm**: Tracks requests in a time window and only allows requests through if they are within the limit
2. **Per-route Configurable Limits**: Each route can have its own rate limit configuration
3. **Storage Abstraction**: In-memory and Redis adapters for storage
4. **HTTP 429 with Headers**: Returns proper HTTP status and headers
5. **TypeScript Types**: Strong typing throughout
6. **Unit Tests**: For the core algorithm and middleware

The system will properly limit the number of requests in a sliding window, store the data in either memory or Redis, and provide appropriate HTTP responses.
